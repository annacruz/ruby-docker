Dockerfile da aplicação

								RUN apt-get update -qq && \\
								apt-get install -y apt-transport-https && \\
								apt-get install -y -qq build-essential ruby-dev libpq-dev && \\ 
								nodejs sendmail

								COPY . /opt/app/
								WORKDIR /opt/app/

								RUN bundle install
								CMD rails s
								EXPOSE 3000

Dockerfile do servidor de emails fake 

							
								FROM ruby:2.3.3

								RUN gem install mailcatcher

								CMD mailcatcher --ip 0.0.0.0 --foreground

E o compose?

							
								version: '2'
								services:
								web:
									build: .
									command:
									rails s -p 3000 -b 0.0.0.0 -e development
									volumes:
									- .:/app/sealevel
									ports:
									- "80:3000"
									networks:
									- backend

								mailcatcher:
									build:
									context: .
									dockerfile: Dockerfile-mailcatcher
									ports:
									- 8080:1080
									networks:
									- backend

								networks:
								backend:
									driver: bridge
